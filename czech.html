library(data.table)

ua=\(x,fun,...){u=unique(x);fun(u,...)[match(x,u)]}
age=\(x,y){x=as.numeric(x);y=as.numeric(y);(y-x-(y-789)%/%1461+(x-789)%/%1461)%/%365}

t0=fread("Czech/data/CR_records.csv")
t=data.table(id=1:nrow(t0),birth=t0$Rok_narozeni,death=t0$DatumUmrti)
t$birth=ua(ua(t$birth,paste0,"-1-1"),as.Date)+sample(0:364,nrow(t),T)
t=t[rep(1:nrow(t),7),]
t$dose=rep(1:7,each=nrow(t0))
t$date=ua(unlist(lapply(1:7,\(i)t0[[paste0("Datum_",i)]])),as.Date,"1970-1-1")
t=t[!is.na(date)][order(-date)]

mindate=min(t$date);maxdate=max(t$death,na.rm=T)

buck=data.table()
for(day in as.list(seq(mindate,maxdate,1))){
  cat(as.character(day),"\n")
  sub=t[date<=day&(is.na(death)|death>=day)]
  sub=unique(sub,by="id") # remove this line to keep people included under previous doses after a new dose
  d=data.table(date=day,dose=sub$dose,week=as.numeric(day-sub$date)%/%7)
  d$age=age(sub$birth,day)
  d$alive=1
  d$dead=nafill(as.numeric(sub$death==day),,0)
  buck=rbind(buck,d)[,.(alive=sum(alive),dead=sum(dead)),by=.(date,dose,week,age)]
}

fwrite(buck,"czbuckets",sep=" ")
