t=fread("nz-record-level-data-4M-records.csv.gz")
redate=\(x){u=unique(x);as.Date(u,"%m-%d-%Y")[match(x,u)]}
t[,c("vax","dead","born"):=lapply(.SD,redate),.SDcols=patterns("date")]

d=rbind(t[order(vax)][,type:=1],t[order(-vax)][,type:=2])
d=d[rowid(type,mrn)==1,.(vax,dead,born,type)]

e1=d[,.(pop=.N),.(date=vax,age=age(born,vax),type)]
e2=d[!is.na(dead),.(pop=-.N),.(date=dead,age=age(born,dead),type)]

e3=d[,.(vax,dead,born,bday=c(outer(born,2021:2023,`year<-`)),type)]
e3=e3[bday>vax&!(!is.na(dead)&bday>dead)]
e3=e3[,.(pop=.N),.(date=bday,age=age(born,bday),type)]
e3=rbind(e3,e3[,.(date,age=age-1,pop=-pop,type)])

e=rbind(e1,e2,e3)[,.(pop=as.double(sum(pop))),.(date,age,type)]
e=merge(do.call(CJ,lapply(e[,-4],unique)),e,all=T)
e[,pop:=cumsum(nafill(pop,,0)),.(age,type)]
